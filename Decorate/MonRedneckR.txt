//These are the Miscellaneous Gibs used by some enemies.
Actor MiscGib1
{
	var int user_t;
	ReactionTime 30
    scale 0.19
    radius 6
    height 5
    speed 10
    +MISSILE
    -NOBLOCKMAP
	+THRUACTORS
    +DOOMBOUNCE
	-NOGRAVITY
    BounceFactor 0.34
    states
    {
    Spawn:
		TNT1 A 0 NoDelay A_SetUserVar(user_t,tid)
		TNT1 A 0 Thing_ChangeTID(0,0)
		TNT1 A 0 ThrustThingZ(0,random(10,40),0,1)
	Loop1:
		RR15 AABBCCDD 2 A_SpawnItemEx("GibBloodTrailMediumSize",random(3,-3),random(3,-3),random(3,-3),0,0,0,0,SXF_TRANSFERTRANSLATION)
		"####" "#" 0 A_Countdown
		Loop
    Death:
		"####" "#" 0 A_JumpIfInventory("ShortGibToken",1,3,AAPTR_PLAYER1)
		"####" "#" 0 A_JumpIf(user_t != 0,2)
		RR15 E 500 
		RR15 E 1 A_FadeOut
		Wait
    }
}
